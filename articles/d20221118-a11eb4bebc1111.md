# コマンド案

```
git branch --list 

git log  --pretty=format:"%ad" -1
git log --date=raw-local --pretty=format:"%ad" -1
date '+%s' --date '14 days ago'  

[[ $(uname -a ) =~ Mac ]]
```

```
  latest_committed_at=$(git log --date=format:'%s' --pretty=format:'%ad' -1)
  cmp=$(date '+%s' --date '14 days ago')
  if [ "$latest_committed_at" -lt "$cmp" ]; then
    echo 
  fi
```
